1 Capturing User Input

Learn how to get text from the user using Entry widgets and the .get() method to retrieve values.

user_input_demo.py 
import tkinter as tk

def show_input():
    """Get text from entry field and display it"""
    user_text = entry.get()  # Get the text from entry widget
    
    if user_text.strip():  # Check if not empty
        result_label.config(
            text=f"You entered: '{user_text}'",
            fg="#27ae60"
        )
    else:
        result_label.config(
            text="Please enter some text first!",
            fg="#e74c3c"
        )

def clear_input():
    """Clear the entry field and result"""
    entry.delete(0, tk.END)  # Clear from position 0 to END
    result_label.config(text="Enter text above and click 'Show Input'")
    entry.focus()  # Put cursor back in entry field

# Create main window

root.configure(bg="#ecf0f1")

title_label.pack(pady=20)

# Instructions
instructions = tk.Label(

    text="Type something in the box below:",
    font=("Arial", 12),

    fg="#34495e"
)
instructions.pack(pady=10)

# Entry widget for user input
entry = tk.Entry(
    root,
    font=("Arial", 14),
    width=30,

)
entry.pack(pady=15)

💡 Understanding User Input
  • tk.Entry() creates a text input field where users can type
  • .get() retrieves the current text from the entry widget
  • .delete(0, tk.END) clears all text from the entry field
  • .focus() places the cursor in the entry field automatically
  • .bind('', function) allows Enter key to trigger actions

2 Processing Numbers and Math

Learn how to convert text input to numbers, perform calculations, and handle errors gracefully.

import tkinter as tk

def add_numbers():
    """Add two numbers with error handling"""
    try:
        # Get text from both entry fields
        num1_text = entry1.get()
        num2_text = entry2.get()
        
        # Convert to numbers (this might cause an error)
        number1 = float(num1_text)
        number2 = float(num2_text)
        
        # Perform the calculation
        result = number1 + number2
        
        # Display the result
        result_label.config(
            text=f"{number1} + {number2} = {result}",
            fg="#27ae60"

        # Handle the case where input isn't a valid number

3 Building a Complete Calculator

Create a professional calculator with multiple operations, advanced features, and a polished interface.

import tkinter as tk
from tkinter import messagebox

def calculate(operation):
    """Perform calculation based on operation type"""
    try:
        # Get numbers from entry fields
        num1 = float(entry1.get())
        num2 = float(entry2.get())
        
        # Perform the requested operation
        if operation == "add":
            result = num1 + num2
            symbol = "+"
        elif operation == "subtract":
            result = num1 - num2
            symbol = "-"
        elif operation == "multiply":
            result = num1 * num2
            symbol = "×"
        elif operation == "divide":
            if num2 == 0:
                result_label.config(
                    text="❌ Cannot divide by zero!",
                    fg="#e74c3c"
                )
                return
            result = num1 / num2
            symbol = "÷"
        
        # Format result nicely
        if result == int(result):
            result_text = f"{num1:g} {symbol} {num2:g} = {int(result)}"
        else:
            result_text = f"{num1:g} {symbol} {num2:g} = {result:.2f}"
        
        result_label.config(text=result_text, fg="#27ae60")
        
        # Update calculation history
        add_to_history(result_text)
        
    except ValueError:
        result_label.config(
            text="❌ Please enter valid numbers!",
            fg="#e74c3c"
        )

def add_to_history(calculation):
    """Add calculation to history list"""
    history_listbox.insert(0, calculation)  # Add to top
    if history_listbox.size() > 10:  # Keep only last 10
        history_listbox.delete(10)

def clear_all():
    """Clear all fields and results"""
    entry1.delete(0, tk.END)
    entry2.delete(0, tk.END)
    result_label.config(
        text="Enter numbers and choose an operation",
        fg="#7f8c8d"
    )
    entry1.focus()

def clear_history():
    """Clear calculation history"""
    history_listbox.delete(0, tk.END)
