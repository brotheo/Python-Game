1 Creating Your First GUI Window

Every desktop application starts with a main window. Let's create one that demonstrates the core concepts of GUI programming.

basic_window.py 
import tkinter as tk

# Create the main application window
root = tk.Tk()
root.title("My First Python GUI Window")
root.geometry("400x300")  # Width x Height in pixels
root.configure(bg="#f0f0f0")  # Light gray background

# Add a welcome message
welcome_label = tk.Label(
    root,
    text="Welcome to GUI Programming!",
    font=("Arial", 16, "bold"),
    bg="#f0f0f0",
    fg="#2c3e50"
)
welcome_label.pack(pady=50)

# Start the GUI event loop (keeps window open)
root.mainloop()
💡 Understanding the Code
  • tk.Tk() creates the main application window object
  • .title() sets the text that appears in the window's title bar
  • .geometry() defines window size as "width x height"
  • .configure(bg=) sets the background color
  • .mainloop() starts the event loop that keeps your window responsive

2 Adding Interactive Elements

Now let's make our application interactive by adding buttons and labels that respond to user actions.

interactive_app.py 
import tkinter as tk

def button_clicked():
    """Function that runs when the button is clicked"""
    status_label.config(
        text="🎉 Button clicked! You're learning GUI programming!",
        fg="#27ae60"
    )
    click_count[0] += 1
    counter_label.config(text=f"Clicks: {click_count[0]}")

def reset_app():
    """Reset the application to initial state"""
    status_label.config(
        text="Click the button above to see magic happen!",
        fg="#34495e"
    )
    click_count[0] = 0
    counter_label.config(text="Clicks: 0")

# Initialize click counter
click_count = [0]

# Create main window
root = tk.Tk()
root.title("Interactive GUI Demo")
root.geometry("500x350")
root.configure(bg="#ecf0f1")

# Title
title_label = tk.Label(
    root,
    text="Interactive Elements Demo",
    font=("Arial", 18, "bold"),
    bg="#ecf0f1",
    fg="#2c3e50"
)
title_label.pack(pady=20)

# Main action button
action_button = tk.Button(
    root,
    text="Click Me!",
    command=button_clicked,
    font=("Arial", 14, "bold"),
    bg="#3498db",
    fg="white",
    padx=30,
    pady=10,
    relief="raised",
    cursor="hand2"
)
action_button.pack(pady=15)

# Status label (shows feedback)
status_label = tk.Label(
    root,
    text="Click the button above to see magic happen!",
    font=("Arial", 12),
    bg="#ecf0f1",
    fg="#34495e",
    wraplength=400
)
status_label.pack(pady=15)

# Click counter
counter_label = tk.Label(
    root,
    text="Clicks: 0",
    font=("Arial", 11),
    bg="#ecf0f1",
    fg="#7f8c8d"
)
counter_label.pack(pady=5)

# Reset button
reset_button = tk.Button(
    root,
    text="Reset",
    command=reset_app,
    font=("Arial", 10),
    bg="#e74c3c",
    fg="white",
    padx=20,
    pady=5,
    cursor="hand2"
)
reset_button.pack(pady=20)

root.mainloop()
🎯 Key Programming Concepts
  • Event Handling: Functions that respond to user interactions
  • Widget Properties: Customizing appearance with colors, fonts, and spacing
  • State Management: Keeping track of data like click counts
  • User Feedback: Providing immediate visual responses to actions

3 Building a Complete Application

Let's create a professional-looking application that combines user input, validation, and multiple interactive features.

personal_greeter.py 
import tkinter as tk
from tkinter import messagebox
import random

def greet_user():
    """Create a personalized greeting with validation"""
    name = name_entry.get().strip()
    
    if not name:
        feedback_label.config(
            text="⚠️ Please enter your name first!",
            fg="#e74c3c"
        )
        name_entry.focus()
        return
    
    if len(name) < 2:
        feedback_label.config(
            text="⚠️ Name must be at least 2 characters long!",
            fg="#e74c3c"
        )
        return
    
    # Random greeting messages
    greetings = [
        f"Hello, {name}! Welcome to Python GUI programming! 🎉",
        f"Great to meet you, {name}! You're doing amazing! ⭐",
        f"Hi there, {name}! Ready to build awesome apps? 🚀",
        f"Welcome aboard, {name}! Let's code something cool! 💻"
    ]
    
    greeting = random.choice(greetings)
    feedback_label.config(text=greeting, fg="#27ae60")
    
    # Update statistics
    global greeting_count
    greeting_count += 1
    stats_label.config(text=f"Greetings given: {greeting_count}")

def clear_form():
    """Clear all input and reset the form"""
    name_entry.delete(0, tk.END)
    feedback_label.config(
        text="Enter your name above and click 'Greet Me!' 👆",
        fg="#34495e"
    )
    name_entry.focus()

def show_about():
    """Show information about the application"""
    messagebox.showinfo(
        "About Personal Greeter",
        "Personal Greeter v1.0\n\n"
        "A simple GUI application built with Python Tkinter.\n"
        "This demonstrates user input, validation, and interactive features.\n\n"
        "Created for learning GUI programming!"
    )

# Initialize statistics
greeting_count = 0

# Create main window
root = tk.Tk()
root.title("Personal Greeter - Python GUI App")
root.geometry("600x500")
root.configure(bg="#2c3e50")

# Main container frame
main_frame = tk.Frame(root, bg="#ecf0f1", padx=40, pady=30)
main_frame.pack(expand=True, fill="both", padx=20, pady=20)

# App title
title_label = tk.Label(
    main_frame,
    text="🌟 Personal Greeter App",
    font=("Arial", 24, "bold"),
    bg="#ecf0f1",
    fg="#2c3e50"
)
title_label.pack(pady=(0, 30))

# Input section
input_frame = tk.Frame(main_frame, bg="#ecf0f1")
input_frame.pack(pady=20)

tk.Label(
    input_frame,
    text="What's your name?",
    font=("Arial", 14),
    bg="#ecf0f1",
    fg="#34495e"
).pack(pady=(0, 10))

name_entry = tk.Entry(
    input_frame,
    font=("Arial", 14),
    width=25,
    relief="solid",
    bd=2,
    justify="center"
)
name_entry.pack(pady=(0, 20))
name_entry.bind('', lambda e: greet_user())  # Enter key support

# Buttons section
button_frame = tk.Frame(main_frame, bg="#ecf0f1")
button_frame.pack(pady=20)

greet_button = tk.Button(
    button_frame,
    text="Greet Me! 👋",
    command=greet_user,
    font=("Arial", 12, "bold"),
    bg="#3498db",
    fg="white",
    padx=25,
    pady=10,
    relief="raised",
    cursor="hand2"
)
greet_button.pack(side=tk.LEFT, padx=10)

clear_button = tk.Button(
    button_frame,
    text="Clear 🔄",
    command=clear_form,
    font=("Arial", 12, "bold"),
    bg="#f39c12",
    fg="white",
    padx=25,
    pady=10,
    relief="raised",
    cursor="hand2"
)
clear_button.pack(side=tk.LEFT, padx=10)

about_button = tk.Button(
    button_frame,
    text="About ℹ️",
    command=show_about,
    font=("Arial", 12, "bold"),
    bg="#9b59b6",
    fg="white",
    padx=25,
    pady=10,
    relief="raised",
    cursor="hand2"
)
about_button.pack(side=tk.LEFT, padx=10)

# Feedback section
feedback_label = tk.Label(
    main_frame,
    text="Enter your name above and click 'Greet Me!' 👆",
    font=("Arial", 13),
    bg="#ecf0f1",
    fg="#34495e",
    wraplength=500,
    justify="center"
)
feedback_label.pack(pady=30)

# Statistics
stats_label = tk.Label(
    main_frame,
    text="Greetings given: 0",
    font=("Arial", 10),
    bg="#ecf0f1",
    fg="#7f8c8d"
)
stats_label.pack(pady=(20, 0))

# Set initial focus
name_entry.focus()

root.mainloop()
